# requirements-styletts2.txt — StyleTTS2 synthesis stack for Kaggle / headless environments
#
# Install order matters (see setup_kaggle.sh):
#   1. apt-get install -y espeak-ng libespeak-ng-dev  (system dep for phonemizer)
#   2. pip install torch==2.1.2 torchaudio==2.1.2 --index-url https://download.pytorch.org/whl/cu121
#   3. pip install -r requirements-styletts2.txt
#   4. pip install "styletts2>=0.1,<2.0" --no-deps
#      (styletts2 pins huggingface_hub<0.20 which conflicts with newer transformers;
#       install it with --no-deps so its transitive dep pin is skipped.
#       All styletts2 runtime deps are explicitly listed in this file instead.)
#   5. conda run python -c "import nltk; nltk.download('punkt_tab', quiet=True)"
#      (NLTK tokenizer data required by styletts2.tts at import time)
#
# All versions validated against Kaggle GPU T4/P100 (CUDA 12.1, Python 3.10).
# Strict pins (marked !) are required for binary/API compatibility; do not relax.
