# requirements-xtts2.txt — XTTS2 synthesis stack for Kaggle / headless environments
#
# Install order matters (see setup_kaggle.sh):
#   1. pip install torch==2.1.2 torchaudio==2.1.2 --index-url https://download.pytorch.org/whl/cu121
#   2. pip install -r requirements-xtts2.txt
#
# numpy, scipy, librosa, soundfile, and numba are intentionally absent here.
# TTS==0.22.0 pulls in compatible versions of those packages automatically.
# Pinning numpy alongside TTS in a single pip pass causes a ResolutionImpossible
# conflict (pip resolves TTS's transitive trainer dep against numpy, and an
# explicit pin to 1.26.4 can violate that range on current PyPI metadata).
#
# Pinned to the exact set validated against Kaggle GPU (CUDA 12.1):
TTS==0.22.0
transformers==4.38.2
tokenizers==0.15.2
accelerate==0.27.2
sentencepiece==0.1.99
